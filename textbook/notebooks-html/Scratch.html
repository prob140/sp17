def jt_probs(r, h):
    x = make_array((1-r)**2, 2*r*(1-r), r**2)
    if r == 0.2:
        return 0.75*x.item(h)
    elif r == 0.9:
        return 0.25*x.item(h)

jt_table = tbl.probability_function(jt_probs)
